### T04 — Implement Slip Nova detection at top of `executeDash()`

**File:** `game.js`

**Location:** `executeDash()` (~line 485). Insert the following block as the **first** executable statement inside the function body, before any position mutation or cooldown reset.

```js
// Slip Nova: if 5+ enemies lie along the dash cylinder, explode instead of teleporting
if (player.slipNova) {
  // Compute destination using same clamping logic as normal dash
  const angle = Math.atan2(mouseY - player.y, mouseX - player.x);
  const dashDist = 160; // match existing dash distance constant
  let destX = player.x + Math.cos(angle) * dashDist;
  let destY = player.y + Math.sin(angle) * dashDist;
  destX = Math.max(player.radius, Math.min(canvas.width  - player.radius, destX));
  destY = Math.max(player.radius, Math.min(canvas.height - player.radius, destY));

  // Count enemies within 30px of the dash line segment
  let novaCount = 0;
  for (const e of enemies) {
    if (pointToSegmentDist(e.x, e.y, player.x, player.y, destX, destY) < 30 + e.radius) {
      novaCount++;
    }
  }

  if (novaCount >= 5) {
    const mx = (player.x + destX) / 2;
    const my = (player.y + destY) / 2;
    triggerExplosion(mx, my, 250, player.damage * 4);
    player.dashCd = 3; // normal dash cooldown; adjust to match existing constant
    player.invuln  = 0.3;
    return; // skip teleport
  }
}
```

**Note on `pointToSegmentDist`:** Search the file for this helper — it may already exist (used by chain lightning or wire logic). If it does not exist, add the following utility function near the top of the file, alongside other small math helpers:
```js
function pointToSegmentDist(px, py, ax, ay, bx, by) {
  const dx = bx - ax, dy = by - ay;
  const lenSq = dx * dx + dy * dy;
  if (lenSq === 0) return Math.hypot(px - ax, py - ay);
  let t = ((px - ax) * dx + (py - ay) * dy) / lenSq;
  t = Math.max(0, Math.min(1, t));
  return Math.hypot(px - (ax + t * dx), py - (ay + t * dy));
}
```

### T06 — Add `updateWireTrails(dt)` and call it from `update()`; confirm stun guard in `updateEnemies()`

**File:** `game.js`

**Location A — new `updateWireTrails(dt)` function.** Add this function anywhere in the file near the other `update*` helpers (e.g. after `updateBullets()` at ~line 1194):

```js
function updateWireTrails(dt) {
  for (let i = wireTrails.length - 1; i >= 0; i--) {
    const w = wireTrails[i];
    w.life -= dt;
    if (w.life <= 0) { wireTrails.splice(i, 1); continue; }
    for (const e of enemies) {
      if (e.stunTimer > 0) continue; // stunned enemies don't interact
      if (pointToSegmentDist(e.x, e.y, w.x1, w.y1, w.x2, w.y2) < 40 + e.radius) {
        // Framerate-independent DPS with per-enemy cooldown to prevent per-frame stacking
        if (!e.wireCd || e.wireCd <= 0) {
          e.hp -= w.damage * dt;
          e.wireCd = 0.1;
          if (e.hp <= 0) killEnemy(e);
        }
      }
      if (e.wireCd > 0) e.wireCd -= dt;
    }
  }
}
```

### T07 — Render wire trails in `render()`

**File:** `game.js`

**Location:** `render()` (~line 1793). Find the call to `drawPlayer()` (~line 1904). Insert the following wire-render block **immediately before** the `drawPlayer()` call so wires appear under the player sprite:

```js
// Render razor wire trails
if (wireTrails.length > 0) {
  for (const w of wireTrails) {
    const alpha = w.life / w.maxLife;
    ctx.save();
    ctx.globalAlpha = alpha * 0.85;
    ctx.strokeStyle = '#00b894';
    ctx.lineWidth = 3 * alpha + 1;
    ctx.shadowColor = '#00b894';
    ctx.shadowBlur = 12;
    ctx.beginPath();
    ctx.moveTo(w.x1, w.y1);
    ctx.lineTo(w.x2, w.y2);
    ctx.stroke();
    ctx.restore();
  }
}
```
