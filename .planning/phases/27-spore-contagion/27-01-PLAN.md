**Block A — Contagion detonation** (inserted as the FIRST block inside the function body, before `kills++` or any other statement — position matters so that an infected enemy explodes before its other death effects):

```js
  // Phase 27: Spore Contagion — detonate if this enemy was infected by a chain
  if (e.infected) {
    triggerExplosion(e.x, e.y, 50, e.infectedDmg);
    spawnParticles(e.x, e.y, '#39d98a', 16, [50, 150]);
    if (blasts.length < 48) {
      blasts.push({ x: e.x, y: e.y, r: 50, life: 0.22, maxLife: 0.22, crit: false });
    }
    // Propagate infection to nearby Spores/Sporelings at 60% decayed damage
    const nextDmg = e.infectedDmg * 0.6;
    if (nextDmg >= 1) {
      for (const nb of enemies) {
        if (nb === e || nb.hp <= 0) continue;
        if (nb.type !== 'spore' && nb.type !== 'sporeling') continue;
        if (Math.hypot(nb.x - e.x, nb.y - e.y) <= 60) {
          nb.infected    = true;
          nb.infectedDmg = nextDmg;
          spawnParticles(nb.x, nb.y, '#39d98a', 5);
        }
      }
    }
  }
```

    if (e.infected) {
