### T09 — Add `drawBountyTarget()` helper function

**File:** `games/neon-swarm/game.js`

**Location:** Add as a new function near other draw helpers (e.g., near `drawPowerupTimers()` at ~line 2470 or `drawPlayer()` at ~line 1904). Place it before `render()`.

**Code to add:**

```js
function drawBountyTarget() {
  if (!bountyTarget || bountyTimer <= 0) return;
  const e = bountyTarget;
  const cx = e.x;
  const cy = e.y;

  // Timer arc under the enemy
  const arcFraction = bountyTimer / 12;
  const arcColor = bountyTimer < 3 ? '#FF4444' : '#FFD700';
  ctx.save();
  ctx.beginPath();
  ctx.arc(cx, cy, e.radius + 8, -Math.PI / 2, -Math.PI / 2 + arcFraction * Math.PI * 2);
  ctx.strokeStyle = arcColor;
  ctx.lineWidth = 2.5;
  ctx.lineCap = 'round';
  ctx.stroke();
  ctx.restore();

  // Pulsing crown (5-point star) above the enemy
  const crownCy = cy - e.radius - 14;
  const pulse = 1 + 0.08 * Math.sin(elapsed * 6);
  const outerR = 10 * pulse;
  const innerR = 4 * pulse;
  const points = 5;
  ctx.save();
  ctx.beginPath();
  for (let i = 0; i < points * 2; i++) {
    const angle = (i * Math.PI) / points - Math.PI / 2;
    const r = i % 2 === 0 ? outerR : innerR;
    const px = cx + r * Math.cos(angle);
    const py = crownCy + r * Math.sin(angle);
    if (i === 0) ctx.moveTo(px, py);
    else ctx.lineTo(px, py);
  }
  ctx.closePath();
  ctx.fillStyle = '#FFD700';
  ctx.fill();
  ctx.strokeStyle = '#FFA500';
  ctx.lineWidth = 1.5;
  ctx.stroke();
  ctx.restore();
}
```

**Notes:**
- `elapsed` is the global elapsed-seconds counter used throughout the draw code (e.g., in `drawPlayer()`). The `elapsed * 6` frequency gives approximately one pulse per ~1 second at visible scale.
- The arc sweeps from 12 o'clock (`-Math.PI/2`) clockwise, proportional to `bountyTimer / 12`.
- The star path uses standard 5-point star geometry: alternating outer and inner radii at evenly spaced angles.
