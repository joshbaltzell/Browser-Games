    ```js
    // PARRY: check if any deflectable eBullet was born within 0.1s and intersects dash path
    let parried = false;
    let parryCx = player.x, parryCy = player.y; // contact point, default to dest
    for (const b of eBullets) {
      if (!b.deflectable) continue;
      if (elapsed - b.birthTime > 0.1) continue;

      // Point-to-segment distance: is bullet center within (player.radius + b.radius) of dash segment?
      const segDx = player.x - ox, segDy = player.y - oy;
      const segLen2 = segDx * segDx + segDy * segDy;
      let t = 0;
      if (segLen2 > 0) {
        t = Math.max(0, Math.min(1, ((b.x - ox) * segDx + (b.y - oy) * segDy) / segLen2));
      }
      const closestX = ox + t * segDx;
      const closestY = oy + t * segDy;
      const dist2 = (b.x - closestX) ** 2 + (b.y - closestY) ** 2;
      const hitRadius = player.radius + b.radius;
      if (dist2 >= hitRadius * hitRadius) continue;

      // Parry confirmed — deflect the bullet
      b.deflectable = false;
      b.deflected = true;
      b.damage *= 3;
      b.playerOwned = true;

      // Reverse velocity toward the owner Sentinel
      const speed = Math.hypot(b.vx, b.vy);
      const toAng = Math.atan2(b.owner.y - b.y, b.owner.x - b.x);
      b.vx = Math.cos(toAng) * speed;
      b.vy = Math.sin(toAng) * speed;

      parryCx = closestX;
      parryCy = closestY;
      parried = true;
    }

    if (parried) {
      // Stun enemies within 80px of contact point
      for (const e of enemies) {
        if (Math.hypot(e.x - parryCx, e.y - parryCy) < 80) {
          e.stunTimer = 0.5;
        }
      }
      // Gold ring burst at contact point
      blasts.push({ x: parryCx, y: parryCy, r: 55, life: 0.28, maxLife: 0.28, crit: true });
      // Gold "PARRY!" floater
      spawnFloater(player.x, player.y - 30, 'PARRY!', COLORS.gold, 24);
      // High-pitched ping
      playTone({ type: "sine", freq: 1200, endFreq: 900, dur: 0.12, gain: 0.14 });
    }
    ```
